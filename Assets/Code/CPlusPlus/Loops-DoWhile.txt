---main.cpp---
#include <iostream>

int main(){

    const int COUNT{10};
    size_t i{0}; // Iterator declaration

    do{
        std::cout << i << " : I love C++" << std::endl;
        ++i; // Incrementation
    }while( i < COUNT);

    std::cout << "Loop done!" << std::endl;


    // Infinite loop
    do{
        std::cout << i  << " : I love C++" << std::endl;
        ++i;
    }while(true);

    return 0;
}
